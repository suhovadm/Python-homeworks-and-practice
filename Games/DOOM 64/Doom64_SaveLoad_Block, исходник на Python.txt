import os
import time

# Путь к папке с сейвами.
save_directory = "C:\DOOM 64 v2.5"

def block_saves():
    while True:
        # Получаем список файлов в директории с сохраненками.
        files = os.listdir(save_directory)

        # Если найдены файлы с сохраненками, удаляем их.
        for file in files:
            # Расширение файлов сохранёнок у Zandronum-a - .zds.
            # ... and not file.startswith("auto") - делаем проверку на автосейв и тем самым не трогаем его.
            # Таким образом, файл будет удалён, если он заканчивается на .zds и НЕ начинается на auto.
            if file.endswith(".zds") and not file.startswith("auto"):
                os.remove(os.path.join(save_directory, file))
                print(f"Delete file: {file}")

        time.sleep(2) # Делаем проверку на удаление каждые 2 секунды.

# Запускаем программу.
if __name__ == "__main__":
    block_saves()
