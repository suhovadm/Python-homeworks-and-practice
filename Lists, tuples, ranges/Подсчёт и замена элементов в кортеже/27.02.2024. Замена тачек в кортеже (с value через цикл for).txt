# Задача: заменить автомашины в кортеже.
# Реализуйте функционал через цикл for.

cars = ('Mercedes', 'Mercedes', 'Mercedes', 'BMW', 'Toyota', 'Honda', 'Subaru')
print(cars) # Заводим кортеж и сразу же выводим на экран.

# Приглашение к вводу команд.
value1 = input('Введите заменяемое название:~$ ')
value2 = input('Введите название НА которое нужно заменить:~$ ')

# Заводим случайную переменную.
a = list(cars) # Перекидываем кортеж в список.

# Запускаем цикл for для списка, который сделали из кортежа выше.
for i in list(cars):

    if i == value1:
        a.remove(value1) # Сначала убираем все значения, которые ввёл пользователь,
        a.insert(0, value2) # затем накидываем те, которые нужны.

print(tuple(a)) # Перекидываем список обратно в кортеж и принтуем на экран.
